    Customized calendar sample with API

    • Result
    • JS
    • HTML 
            var cal = new tui.component.Calendar({
            element: '#layer',
            classPrefix: 'calendar2-', // class prefix for customizing
            year: 2016,
            month: 10,
            titleFormat: 'yyyy / mm',
            todayFormat: 'yyyy / mm / dd'
        });

        cal.on('draw', function(drawData) {
            var today = new Date();
            var isToday = true;
            var isTodayMonth = true;
            var isTodayYear = true;

            if (drawData.year !== today.getFullYear()) {
                isToday = false;
                isTodayYear = false;
            }

            if (drawData.month !== (today.getMonth() + 1)) {
                isToday = false;
                isTodayMonth = false;
            }

            if (drawData.date !== (today.getDate())) {
                isToday = false;
            }

            if (isToday || isTodayMonth || isTodayYear) {
                drawData.$date.html('<span class="today">' + drawData.html + '</span>');
            }
        });
        cal.draw();
